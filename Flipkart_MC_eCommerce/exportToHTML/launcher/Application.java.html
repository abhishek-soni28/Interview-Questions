package launcher;

import repository.InMemoryRepository;
import services.BookingSystem;

public class Application {
    public static void main(String[] args) {
        // Initialize the in-memory repository and the product system
        InMemoryRepository repository = new InMemoryRepository();
        BookingSystem bookingSystem = new BookingSystem(repository);

        // Admin operations
        System.out.println("Admin operations:");

        // Add Venue
        System.out.println("Adding Venue...");
        bookingSystem.addVenue("v_id_001", "SarjapurSocial", "Bengaluru", 5);
        bookingSystem.addVenue("v_id_002", "Runway47", "Bengaluru", 5);

        // Add Event
        System.out.println("Adding Event.");
        bookingSystem.addEvent("e_id_001", "v_id_001", "MaiSamayHoon", "StandUp", 4, "19:00", "21:00", 500);
        bookingSystem.addEvent("e_id_002", "v_id_001", "MaiSamayHoon", "StandUp", 4, "19:00", "21:00", 500);

        // Customer operations
        System.out.println("\nCustomer operations:");

        // Book Event
        System.out.println("Customer product event.");
        bookingSystem.bookEvent("b_id_001", "c_id_001", "e_id_001");
        bookingSystem.bookEvent("b_id_002", "c_id_002", "e_id_001");
        bookingSystem.bookEvent("b_id_003", "c_id_003", "e_id_001");
        bookingSystem.bookEvent("b_id_004", "c_id_004", "e_id_001");
        bookingSystem.bookEvent("b_id_005", "c_id_005", "e_id_001");
        bookingSystem.cancelBooking("c_id_002", "b_id_002");

        // removing venue
        bookingSystem.removeVenue("v_id_001");
        //product removed venue
        bookingSystem.bookEvent("b_id_005", "c_id_005", "e_id_001");
        //adding irrevelant event
        bookingSystem.bookEvent("b_id_005", "c_id_005", "e_id_004");
        //removing event
        bookingSystem.removeEvent("e_id_002");

//
//        // Get All Bookings (for customer)
//        System.out.println("Fetching all products for customer c_id_001...");
//        bookingSystem.searchByEventType("StandUp", "startTime");
    }
}
